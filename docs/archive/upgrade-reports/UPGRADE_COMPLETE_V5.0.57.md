### 5.2 核心算法伪代码

```javascript
// 预测误差计算
function calculatePredictionError(sensoryInput, prediction, precision) {
  const error = sensoryInput - prediction;
  const weightedError = error * precision;
  return { error, weightedError, precision };
}

// 自由能计算
function calculateFreeEnergy(predictionError, modelComplexity) {
  const accuracy = -0.5 * Math.pow(predictionError, 2);
  const complexity = modelComplexity * LOG_2_PI;
  return accuracy - complexity;
}

// 主动推理策略选择
function selectPolicy(policies, expectedFreeEnergy) {
  const softmax = (efe) => Math.exp(-efe) / sum(exp(-efe));
  const probabilities = policies.map(p => softmax(expectedFreeEnergy[p]));
  return weightedRandomSelect(policies, probabilities);
}

// 生成模型更新
function updateGenerativeModel(model, predictionError, learningRate) {
  const gradient = calculateGradient(model, predictionError);
  return model + learningRate * gradient;
}
```
