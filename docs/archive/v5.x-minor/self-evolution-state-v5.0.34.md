### 核心算法更新

```javascript
// 五层预测架构
class FiveLayerPredictiveArchitecture {
  constructor() {
    this.layers = {
      interoceptive: new InteroceptivePrediction(),
      affective: new AffectivePrediction(),
      perceptual: new PerceptualPrediction(),
      social: new SocialPrediction(),
      metacognitive: new MetacognitivePrediction()
    };
  }

  // 前反思自我意识 (第 1-2 层)
  getPrereflectiveSelf() {
    return {
      interoceptive: this.layers.interoceptive.getPrediction(),
      affective: this.layers.affective.getPrediction()
    };
  }

  // 反思自我意识 (第 3-5 层)
  getReflectiveSelf() {
    return {
      perceptual: this.layers.perceptual.getPrediction(),
      social: this.layers.social.getPrediction(),
      metacognitive: this.layers.metacognitive.getPrediction()
    };
  }
}

// 时间自我三维度
class TemporalSelfIntegration {
  constructor() {
    this.past = new AutobiographicalPast();
    this.present = new PhenomenalPresent();
    this.future = new ProspectiveFuture();
  }

  // 时间自我连续性计算
  calculateContinuity() {
    const pastPresentCoherence = this.calculateCoherence(this.past, this.present);
    const presentFutureCoherence = this.calculateCoherence(this.present, this.future);
    const pastFutureCoherence = this.calculateCoherence(this.past, this.future);
    
    return (pastPresentCoherence + presentFutureCoherence + pastFutureCoherence) / 3;
  }

  // 时间自我断裂检测
  detectDiscontinuity() {
    const continuity = this.calculateContinuity();
    return continuity < 0.6; // 阈值以下表示断裂
  }
}

// 他心预测误差 (OMPE) 计算
function calculateOMPE(expectedOtherState, actualOtherState, empathyAccuracy) {
  const behaviorError = Math.abs(expectedOtherState.behavior - actualOtherState.behavior);
  const intentionError = Math.abs(expectedOtherState.intention - actualOtherState.intention);
  const affectiveError = Math.abs(expectedOtherState.affect - actualOtherState.affect);
  
  const OMPE = (behaviorError * 0.3 + intentionError * 0.4 + affectiveError * 0.3) * empathyAccuracy;
  return { OMPE, behaviorError, intentionError, affectiveError };
}

// 共情精度加权 (EPW)
function calculateEPW(empathySignal, historicalAccuracy) {
  const currentConfidence = empathySignal.confidence;
  const historicalWeight = historicalAccuracy * 0.5;
  const EPW = currentConfidence * 0.5 + historicalWeight;
  return EPW;
}

// 社会模型更新率 (SMUR)
function calculateSMUR(OMPE, EPW, socialLearningRateBase) {
  const updateRate = socialLearningRateBase * OMPE * EPW;
  return Math.min(updateRate, 0.25); // 限制最大更新率
}
```
