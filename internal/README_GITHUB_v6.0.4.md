## 🔧 Usage | 使用

### Basic Usage | 基本使用

```javascript
const heartflow = require('heartflow-companion');

// Initialize | 初始化
const hf = new heartflow.HeartFlow({
  language: 'zh-CN', // or 'en-US'
  personalityTracking: true
});

// Analyze emotion | 分析情绪
const result = await hf.analyzeEmotion('我感到很焦虑', {
  context: 'work',
  intensity: 75
});

console.log(result);
```
