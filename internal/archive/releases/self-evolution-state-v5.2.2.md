class HierarchicalMetacognitionMonitor_v5_2_2 {
    constructor() {
        this.levels = {
            L1: new ObjectLevelMonitor(),      // Task performance
            L2: new StrategyLevelMonitor(),    // Strategy effectiveness
            L3: new CapabilityLevelMonitor(),  // Metacognitive accuracy
            L4: new FrameworkLevelMonitor()    // Framework applicability
        };
        this.errorPropagationThreshold = 0.7;
        this.interventionGenerator = new InterventionGenerator();
    }

    monitor(cognitiveProcess) {
        const monitoringResults = {};
        const errors = [];
        
        // L1: Object-level monitoring
        monitoringResults.L1 = this.levels.L1.evaluate(cognitiveProcess);
        if (monitoringResults.L1.error > this.errorPropagationThreshold) {
            errors.push({level: 'L1', ...monitoringResults.L1});
        }
        
        // L2: Strategy monitoring (activated on L1 error)
        if (errors.length > 0) {
            monitoringResults.L2 = this.levels.L2.evaluate(
                cognitiveProcess.strategy
            );
            if (monitoringResults.L2.error > this.errorPropagationThreshold) {
                errors.push({level: 'L2', ...monitoringResults.L2});
            }
        }
        
        // L3: Capability monitoring
        monitoringResults.L3 = this.levels.L3.evaluate(
            this.getMetacognitiveAccuracy()
        );
        
        // L4: Framework monitoring
        monitoringResults.L4 = this.levels.L4.evaluate(
            this.getCognitiveFramework()
        );
        
        // Generate intervention if needed
        const intervention = errors.length > 0 
            ? this.interventionGenerator.generate(errors)
            : null;
        
        return {
            monitoringResults,
            errors,
            interventionNeeded: errors.length > 0,
            intervention,
            metacognitiveAccuracy: this.calculateOverallAccuracy()
        };
    }
}
